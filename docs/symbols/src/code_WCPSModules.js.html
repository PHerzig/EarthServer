 92 /**
 93  * Creates the x3d geometry and appends it to the given root node. This is done automatically by the SceneManager.
 94  * @param root - X3D node to append the model.
 95  * @param index - Index of this model in the SceneManager list.
 96  * @param cubeSizeX - Size of the fishtank/cube on the x-axis.
 97  * @param cubeSizeY - Size of the fishtank/cube on the y-axis.
 98  * @param cubeSizeZ - Size of the fishtank/cube on the z-axis.
 99  */
100 EarthServerGenericClient.Model_WCPSDemAlpha.prototype.createModel=function(root, index, cubeSizeX, cubeSizeY, cubeSizeZ){
101     if( root === undefined)
102         alert("root is not defined");
103 
104     this.cubeSizeX = cubeSizeX;
105     this.cubeSizeY = cubeSizeY;
106     this.cubeSizeZ = cubeSizeZ;
107 
108     this.root = root;
109     this.index = index;
110 
111     //1: Check if mandatory values are set
112     if( this.coverageImage === undefined || this.coverageDEM === undefined || this.url === undefined
113         || this.minx === undefined || this.miny === undefined || this.maxx === undefined || this.maxy === undefined )
114     {
115         alert("Not all mandatory values are set. WCPSDemAlpha: " + this.name );
116         alert(this.coverageImage + this.coverageDEM + this.url + this.minx === undefined + this.miny+ this.maxx + this.maxy );
117         console.log(this);
118         return;
119     }
120 
121     //2: create wcps query
122     //Either user set if query strings are set for all wcps channels or standard wcps query if wcps channels are not set
123     //IF something is not defined use standard query.
124     if( this.WCPSString[0] === undefined || this.WCPSString[1] === undefined || this.WCPSString[2] === undefined || this.WCPSString[3] === undefined)
125     {
126         this.wcpsQuery =  "for i in (" + this.coverageImage + "), dtm in (" + this.coverageDEM + ") return encode ( { ";
127         this.wcpsQuery += "red: scale(trim(i.red, {x(" + this.minx + ":" +  this.maxx + "), y(" + this.miny + ":" + this.maxy + ') }), {x:"CRS:1"(0:' + this.XResolution + '), y:"CRS:1"(0:' + this.ZResolution + ")}, {}); ";
128         this.wcpsQuery += "green: scale(trim(i.green, {x(" + this.minx + ":" +  this.maxx + "), y(" + this.miny + ":" + this.maxy + ') }), {x:"CRS:1"(0:' + this.XResolution + '), y:"CRS:1"(0:' + this.ZResolution + ")}, {}); ";
129         this.wcpsQuery += "blue: scale(trim(i.blue, {x(" + this.minx + ":" +  this.maxx + "), y(" + this.miny + ":" + this.maxy + ') }), {x:"CRS:1"(0:' + this.XResolution + '), y:"CRS:1"(0:' + this.ZResolution + ")}, {});";
130         this.wcpsQuery += "alpha: (char) (((scale(trim(dtm , {x(" + this.minx + ":" +  this.maxx + "), y(" + this.miny + ":" + this.maxy + ') }), {x:"CRS:1"(0:' + this.XResolution + '), y:"CRS:1"(0:' + this.ZResolution + ")}, {})) / 1349) * 255)";
131         this.wcpsQuery += '}, "' + this.imageFormat +'" )';
132     }
133     else //ALL set so use custom query
134     {
135         //Replace $ symbols with the actual values
136         for(var i=0; i<4; i++)
137         {
138             this.WCPSString[i] = this.WCPSString[i].replace("$CI","image");
139             this.WCPSString[i] = this.WCPSString[i].replace("$CD","dtm");
140             this.WCPSString[i] = this.WCPSString[i].replace("$MINX",this.minx);
141             this.WCPSString[i] = this.WCPSString[i].replace("$MINY",this.miny);
142             this.WCPSString[i] = this.WCPSString[i].replace("$MAXX",this.maxx);
143             this.WCPSString[i] = this.WCPSString[i].replace("$MAXY",this.maxy);
144             this.WCPSString[i] = this.WCPSString[i].replace("$RESX",this.XResolution);
145             this.WCPSString[i] = this.WCPSString[i].replace("$RESZ",this.ZResolution);
146         }
147         this.wcpsQuery =  "for image in (" + this.coverageImage + "), dtm in (" + this.coverageDEM + ") return encode ( { ";
148         this.wcpsQuery += "red: " + this.WCPSString[0] + " ";
149         this.wcpsQuery += "green: " + this.WCPSString[1]+ " ";
150         this.wcpsQuery += "blue: " + this.WCPSString[2] + " ";
151         this.wcpsQuery += "alpha: " + this.WCPSString[3];
152         this.wcpsQuery += '}, "' + this.imageFormat +'" )';
153     }
154 
155     //3: Make ServerRequest and receive data.
156     var data = {
157         wcpsUrl:this.url,
158         wcpsQuery:this.wcpsQuery,
159         useWcs:false
160     };
161     var serverRequest = new ServerRequest(this);
162     console.time(this.name+"_request");
163     serverRequest.request(data);
164 };
165 /**
166  * This is a callback method as soon as the ServerRequest in createModel() has received it's data.
167  * This is done automatically.
168  * @param data - Received data from the ServerRequest.
169  */
170 EarthServerGenericClient.Model_WCPSDemAlpha.prototype.receiveData = function( data)
171 {
172     console.timeEnd(this.name+"_request");
173     if( data === null)
174     { console.log("WCPSDemAlpha: Request not successful.");}
175     else
176     {
177         var YResolution = (parseFloat(data.maxMSAT) - parseFloat(data.minMSAT) );
178         var transform = this.createTransform(this.cubeSizeX,this.cubeSizeY,this.cubeSizeZ,data.width,YResolution,data.height,parseFloat(data.minMSAT));
179         this.root.appendChild( transform);
180 
181         //Set transparency
182         data.transparency =  this.transparency;
183 
184         //Create Terrain out of the received data
185         this.terrain = new Terrain(transform, data, this.index);
186         this.terrain.createTerrain();
187     }
188 };
