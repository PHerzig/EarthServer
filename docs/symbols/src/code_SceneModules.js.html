  9 Function.prototype.inheritsFrom = function( parentClassOrObject )
 10 {
 11     if ( parentClassOrObject.constructor == Function )
 12     {
 13         //Normal Inheritance
 14         this.prototype = new parentClassOrObject;
 15         this.prototype.constructor = this;
 16         this.prototype.parent = parentClassOrObject.prototype;
 17     }
 18     else
 19     {
 20         //Pure Virtual Inheritance
 21         this.prototype = parentClassOrObject;
 22         this.prototype.constructor = this;
 23         this.prototype.parent = parentClassOrObject;
 24     }
 25     return this;
 26 };

233     /**
234      * Update Offset changes the position of the current selected SceneModel on the x-,y- or z-Axis.
235      * @param which - Which Axis will be changed (0:X 1:Y 2:Z)
236      * @param value - The new position
237      */
238     this.updateOffset = function(which,value)
239     {
240         var trans = document.getElementById("EarthServerGenericClient_modelTransform"+this.currentUIElement);
241 
242         if( trans )
243         {
244             var offset;
245             switch(which)
246             {
247                 case 0: offset = this.cubeSizeX/2.0; break;
248                 case 1: offset = this.cubeSizeY/2.0; break;
249                 case 2: offset = this.cubeSizeZ/2.0; break;
250             }
251             var oldTrans = trans.getAttribute("translation");
252             oldTrans = oldTrans.split(" ");
253             oldTrans[which] = value - offset;
254             //alert(oldTrans);
255             trans.setAttribute("translation",oldTrans[0] + " " + oldTrans[1] + " " + oldTrans[2]);
256         }
257     };
258 
259     /**
260      * This changes the scaling on the Y-Axis(Elevation).
261      * @param value - The base elevation is multiplied by this value
262      */
263     this.updateElevation =function(value)
264     {
265         var trans = document.getElementById("EarthServerGenericClient_modelTransform"+this.currentUIElement);
266 
267         if( trans )
268         {
269             var oldTrans = trans.getAttribute("scale");
270             oldTrans = oldTrans.split(" ");
271 
272             if( this.baseElevation[this.currentUIElement] === undefined)
273             {
274                 this.baseElevation[this.currentUIElement] = oldTrans[1];
275             }
276 
277             oldTrans[1] = value*this.baseElevation[this.currentUIElement]/10;
278 
279             trans.setAttribute("scale",oldTrans[0] + " " + oldTrans[1] + " " + oldTrans[2]);
280         }
281     };

404 /**
405  * @class Abstract base class for scene models.
406  */
407 EarthServerGenericClient.AbstractSceneModel = function(){
408     /**
409      * Sets the name of the scene model to be displayed.
410      * @param modelName - Name of the model
411      */
412     this.setName = function(modelName){
413         this.name = String(modelName);
414     };
415     /**
416      * Sets the area of interest for the model. (Lower Corner, Upper Corner)
417      * @param minx - Minimum/Lower Latitude
418      * @param miny - Minimum/Lower Longitude
419      * @param maxx - Maximum/Upper Latitude
420      * @param maxy - Maximum/Upper Longitude
421      */
422     this.setAreaOfInterest = function(minx,miny,maxx,maxy){
423         this.minx = minx;
424         this.miny = miny;
425         this.maxx = maxx;
426         this.maxy = maxy;
427     };
428     /**
429      * Sets the resolution of the scene model (if possible).
430      * @param xRes - Resolution on the x-axis/Latitude
431      * @param zRes - Resolution on the z-axis/Longitude
432      */
433     this.setResolution = function(xRes,zRes){
434         this.XResolution = parseInt(xRes);
435         this.ZResolution = parseInt(zRes);
436 
437         var maxResolution = EarthServerGenericClient_MainScene.getMaxResolution();
438         if( this.XResolution > maxResolution )
439         {   this.XResolution = maxResolution;   }
440         if( this.ZResolution > maxResolution )
441         {   this.ZResolution = maxResolution;   }
442 
443     };
444     /**
445      * Sets the names of the axes to be displayed [Not used yet]
446      * @param xLabel - width
447      * @param yLabel - height
448      * @param zLabel - depth
449      */
450     this.setAxisLabels = function( xLabel, yLabel, zLabel){
451         this.xLabel = String(xLabel);
452         this.yLabel = String(yLabel);
453         this.zLabel = String(zLabel);
454     };
455     /**
456      * Sets the position of the scene model within the fishtank/cube. Values between [0-1]
457      * @param xOffset - Offset on the x-axis/width  Default:0
458      * @param yOffset - Offset on the y-axis/height Default:0
459      * @param zOffset - Offset on the z-axis/depth  Default:0
460      */
461     this.setOffset = function( xOffset, yOffset, zOffset){
462         this.xOffset = parseFloat(xOffset);
463         this.yOffset = parseFloat(yOffset);
464         this.zOffset = parseFloat(zOffset);
465     };
466     /**
467      * Sets the size of the scene model compared to the fishtank/cube. Values between 0 - 1.
468      * @param xScale - Size of the model on the x-axis/width  Default:1   (whole cube)
469      * @param yScale - Size of the model on the y-axis/height Default:0.3 (30% of the cube)
470      * @param zScale - Size of the model on the x-axis/width  Default:1   (whole cube)
471      */
472     this.setScale = function( xScale, yScale, zScale){
473         this.xScale = parseFloat(xScale);
474         this.yScale = parseFloat(yScale);
475         this.zScale = parseFloat(zScale);
476     };
477 
478     /**
479      * Sets the image format for the server request.
480      * @param imageFormat - Default "png"
481      */
482     this.setImageFormat = function( imageFormat){
483         this.imageFormat = String(imageFormat);
484     };
485 
486     /**
487      * Sets the transparency of the scene model. Values between 0-1 (Fully Opaque - Fully Transparent).
488      * @param transparency
489      */
490     this.setTransparency = function( transparency ){
491         this.transparency = parseFloat(transparency);
492     };
493 
494     /**
495      * Creates the transform for the scene model to fit into the fishtank/cube. This is done automatically by
496      * the scene model.
497      * @param cubeSizeX - Size of the fishtank/cube on the x-axis
498      * @param cubeSizeY - Size of the fishtank/cube on the y-axis
499      * @param cubeSizeZ - Size of the fishtank/cube on the z-axis
500      * @param XRes - Size of the received data on the x-axis (e.g. the requested DEM )
501      * @param YRes - Size of the received data on the y-axis
502      * @param ZRes - Size of the received data on the z-axis
503      * @param minvalue - Minumum Value of the along the y-axis (e.g. minimum value in a DEM, so the model starts at it's wished location)
504      * @return {Element}
505      */
506     this.createTransform = function(cubeSizeX,cubeSizeY,cubeSizeZ,XRes,YRes,ZRes,minvalue){
507         var trans = document.createElement('Transform');
508         trans.setAttribute("id", "EarthServerGenericClient_modelTransform"+this.modelID);
509         var scaleX,scaleY,scaleZ;
510 
511         //console.log(XRes + "/" + YRes + "/" + ZRes);
512 
513         this.YResolution = YRes;
514 
515         scaleX = (cubeSizeX*this.xScale)/(parseInt(XRes)-1);
516         scaleY = (cubeSizeY*this.yScale)/this.YResolution;
517         scaleZ = (cubeSizeZ*this.zScale)/(ZRes-1);
518 
519         trans.setAttribute("scale", "" + scaleX + " " + scaleY + " " + scaleZ);
520 
521         var xoff = (cubeSizeX * this.xOffset) - (cubeSizeX/2.0);
522         var yoff = (cubeSizeY * this.yOffset) - (minvalue*scaleY) - (cubeSizeY/2.0);
523         var zoff = (cubeSizeZ * this.zOffset) - (cubeSizeZ/2.0);
524         trans.setAttribute("translation", "" + xoff+ " " + yoff  + " " + zoff);
525
